Class: VectorInt

VectorInt

new VectorInt()

Custom container class for numbers.

Sample usage: 


function onTrack()
{
    //Call track on a frame
    trackerReturnState = m_Tracker.track(
       mWidth, mHeight, ppixels, faceData,
       Module.VisageTrackerImageFormat.VISAGE_FRAMEGRABBER_FMT_RGBA.value,
       Module.VisageTrackerOrigin.VISAGE_FRAMEGRABBER_ORIGIN_TL.value);
    //Test result
    if (trackerReturnState===Module.VisageTrackerStatus.TRACK_STAT_OK.value)
    {
       //Read vertices
       var vertices = [];
       var faceModelVertices = faceData.getFaceModelVertices();
       for (var i = 0; i < faceData.faceModelVertexCount; ++i)
       {
          vertices.push(faceModelVertices.get(i));
       }
       //Clean-up
       faceModelVertices.delete();
    }
}
